ℹ️
실행 방법
MySQL 클라이언트(mysql -u root -p) 또는 DBeaver, TablePlus 등 GUI 툴에서 아래 SQL 전체를 복사해 한 번에 실행하세요. mydb 데이터베이스가 자동 생성됩니다.

-- ────────────────────────────────────────────
-- 학습용 데이터베이스 생성
-- ────────────────────────────────────────────
CREATE DATABASE IF NOT EXISTS mydb
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;

USE mydb;

-- ────────────────────────────────────────────
-- 04, 05, 06, 07, 12, 13 섹션에서 사용
-- 기본 사용자 테이블
-- ────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS users (
    id         BIGINT UNSIGNED  NOT NULL AUTO_INCREMENT,
    name       VARCHAR(100)     NOT NULL,
    email      VARCHAR(255)     NOT NULL,
    age        INT              NOT NULL DEFAULT 0,
    status     VARCHAR(20)      NOT NULL DEFAULT 'active',
    created_at DATETIME         NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    UNIQUE KEY uq_email (email)   -- 12섹션 중복키 에러(1062) 예제용
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ────────────────────────────────────────────
-- 09 섹션에서 사용
-- NULL 허용 컬럼이 포함된 프로필 테이블
-- ────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS profiles (
    id       BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    name     VARCHAR(100)    NOT NULL,
    bio      TEXT                    NULL,   -- sql.NullString
    age      INT                     NULL,   -- sql.NullInt64
    score    DOUBLE                  NULL,   -- sql.NullFloat64
    verified TINYINT(1)              NULL,   -- sql.NullBool
    birthday DATETIME                NULL,   -- sql.NullTime
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ────────────────────────────────────────────
-- 08 섹션에서 사용
-- 계좌 이체 트랜잭션 예제용 테이블
-- ────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS accounts (
    id      BIGINT UNSIGNED    NOT NULL AUTO_INCREMENT,
    name    VARCHAR(100)       NOT NULL,
    balance DECIMAL(15, 2)     NOT NULL DEFAULT 0.00,
    PRIMARY KEY (id),
    CONSTRAINT chk_balance CHECK (balance >= 0)  -- 잔액 음수 방지
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ────────────────────────────────────────────
-- 08 섹션에서 사용
-- 이체 이력 로그 테이블
-- ────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS tx_log (
    id         BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    from_id    BIGINT UNSIGNED NOT NULL,
    to_id      BIGINT UNSIGNED NOT NULL,
    amount     DECIMAL(15, 2)  NOT NULL,
    created_at DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    CONSTRAINT fk_txlog_from FOREIGN KEY (from_id) REFERENCES accounts(id),
    CONSTRAINT fk_txlog_to   FOREIGN KEY (to_id)   REFERENCES accounts(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ────────────────────────────────────────────
-- 08 섹션에서 사용
-- 주문 / 주문 상품 테이블 (WithTx 예제)
-- ────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS orders (
    id         BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id    BIGINT UNSIGNED NOT NULL,
    total      DECIMAL(15, 2)  NOT NULL DEFAULT 0.00,
    created_at DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    CONSTRAINT fk_orders_user FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS order_items (
    id         BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    order_id   BIGINT UNSIGNED NOT NULL,
    product_id BIGINT UNSIGNED NOT NULL,
    qty        INT             NOT NULL DEFAULT 1,
    PRIMARY KEY (id),
    CONSTRAINT fk_items_order FOREIGN KEY (order_id) REFERENCES orders(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

01

개요 및 아키텍처

Go의 database/sql 패키지는 SQL 데이터베이스를 위한 범용 인터페이스를 제공합니다. 패키지 자체는 특정 데이터베이스에 종속되지 않으며, 드라이버(driver)를 통해 실제 DB와 통신합니다. MySQL 연동을 위해서는 go-sql-driver/mysql 드라이버를 함께 사용합니다.

🔌
드라이버 인터페이스
database/sql/driver 인터페이스를 구현한 드라이버를 blank import로 등록합니다.
🏊
자동 커넥션 풀
sql.DB는 내부적으로 커넥션 풀을 관리합니다. 쿼리마다 커넥션을 열고 닫을 필요가 없습니다.
🔒
고루틴 안전
sql.DB는 여러 고루틴에서 동시에 안전하게 사용할 수 있습니다. 전역 변수로 선언해 재사용합니다.
📦
표준 인터페이스
동일한 코드로 MySQL, PostgreSQL, SQLite 등 드라이버만 교체해 DB를 전환할 수 있습니다.

기본 연결
main.go go 
package main

import (
    "database/sql"
    "fmt"
    "log"

    // 드라이버를 blank import로 등록 (init() 함수 실행)
    _ "github.com/go-sql-driver/mysql"
)

var db *sql.DB

func main() {
    var err error

    // DSN 형식: user:password@tcp(host:port)/dbname?params
    dsn := "root:password@tcp(127.0.0.1:3306)/mydb?charset=utf8mb4&parseTime=True&loc=Local"

    // sql.Open은 실제 연결하지 않고 DB 핸들만 생성
    db, err = sql.Open("mysql", dsn)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // Ping으로 실제 연결 확인
    if err = db.Ping(); err != nil {
        log.Fatal("DB 연결 실패:", err)
    }

    fmt.Println("✓ MySQL 연결 성공")
}
DSN (Data Source Name) 파라미터
파라미터예시설명
charsetutf8mb4문자셋 설정. 이모지 지원 시 utf8mb4 사용
parseTimeTrueTIME/DATE/DATETIME을 time.Time으로 자동 파싱
locLocal타임존 설정 (Local, UTC, Asia/Seoul)
timeout5s연결 타임아웃
readTimeout30s읽기 타임아웃
writeTimeout30s쓰기 타임아웃
multiStatementstrue다중 쿼리 허용 (보안 주의)
tlstrueTLS 암호화 연결 사용
ℹ️
sql.Open vs db.Ping
sql.Open()은 드라이버 유효성만 확인하고 실제 연결은 하지 않습니다. 실제 연결을 즉시 확인하려면 반드시 db.Ping() 또는 db.PingContext(ctx)를 호출하세요.
03

커넥션 풀 설정

sql.DB는 내부적으로 커넥션 풀을 관리합니다. 적절한 풀 설정은 성능과 MySQL 서버의 max_connections 한도를 고려해 튜닝해야 합니다.

db.go go 
func NewDB(dsn string) (*sql.DB, error) {
    db, err := sql.Open("mysql", dsn)
    if err != nil {
        return nil, err
    }

    // 최대 열린 커넥션 수 (MySQL max_connections 미만으로 설정)
    db.SetMaxOpenConns(25)

    // 최대 유휴(idle) 커넥션 수
    db.SetMaxIdleConns(10)

    // 커넥션 최대 재사용 시간 (MySQL wait_timeout 이하로 설정)
    db.SetConnMaxLifetime(time.Minute * 3)

    // 유휴 커넥션 최대 대기 시간 (Go 1.15+)
    db.SetConnMaxIdleTime(time.Minute)

    if err = db.Ping(); err != nil {
        return nil, err
    }
    return db, nil
}

// 풀 상태 모니터링
func printDBStats(db *sql.DB) {
    stats := db.Stats()
    fmt.Printf("OpenConnections: %d\n", stats.OpenConnections)
    fmt.Printf("InUse: %d\n", stats.InUse)
    fmt.Printf("Idle: %d\n", stats.Idle)
    fmt.Printf("WaitCount: %d\n", stats.WaitCount)
}
메서드기본값권장 설정설명
SetMaxOpenConns0 (무제한)25동시 열린 커넥션 최대 수. 0은 무제한
SetMaxIdleConns210풀에 유지할 유휴 커넥션 수
SetConnMaxLifetime0 (무제한)3m커넥션 재사용 최대 시간
SetConnMaxIdleTime0 (무제한)1m유휴 커넥션 유지 최대 시간 (Go 1.15+)
⚠️
MySQL wait_timeout 주의
MySQL의 wait_timeout(기본 8시간)이 지나면 서버가 커넥션을 끊습니다. SetConnMaxLifetime을 wait_timeout보다 짧게 설정해 "invalid connection" 에러를 방지하세요.
04

SELECT 쿼리

다중 행 조회 — db.Query()
query.go go 
type User struct {
    ID    int
    Name  string
    Email string
    Age   int
}

func getUsers(db *sql.DB) ([]User, error) {
    // ? 플레이스홀더로 SQL Injection 방지
    rows, err := db.Query(
        "SELECT id, name, email, age FROM users WHERE age > ? ORDER BY id",
        18,
    )
    if err != nil {
        return nil, fmt.Errorf("getUsers: %w", err)
    }
    defer rows.Close() // 반드시 닫아야 커넥션이 풀로 반환됨

    var users []User
    for rows.Next() {
        var u User
        if err := rows.Scan(&u.ID, &u.Name, &u.Email, &u.Age); err != nil {
            return nil, fmt.Errorf("getUsers scan: %w", err)
        }
        users = append(users, u)
    }

    // 반복 중 발생한 에러 확인 (필수!)
    if err = rows.Err(); err != nil {
        return nil, fmt.Errorf("getUsers rows: %w", err)
    }
    return users, nil
}
단일 행 조회 — db.QueryRow()
query.go go 
func getUserByID(db *sql.DB, id int) (*User, error) {
    var u User

    // QueryRow는 항상 *sql.Row를 반환 (에러는 Scan에서 확인)
    err := db.QueryRow(
        "SELECT id, name, email, age FROM users WHERE id = ?",
        id,
    ).Scan(&u.ID, &u.Name, &u.Email, &u.Age)

    switch {
    case errors.Is(err, sql.ErrNoRows):
        // 결과 없음 — nil 반환 또는 커스텀 에러
        return nil, fmt.Errorf("user id=%d not found", id)
    case err != nil:
        return nil, fmt.Errorf("getUserByID: %w", err)
    }
    return &u, nil
}
🚫
rows.Close() 누락은 커넥션 누수!
rows.Close()를 호출하지 않으면 해당 커넥션이 풀로 반환되지 않아 커넥션이 고갈됩니다. 항상 defer rows.Close()를 Query 바로 다음 줄에 작성하세요.

INSERT / UPDATE / DELETE

db.Exec() — 단일 실행
exec.go go 
// INSERT — LastInsertId 활용
func createUser(db *sql.DB, name, email string, age int) (int64, error) {
    result, err := db.Exec(
        "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
        name, email, age,
    )
    if err != nil {
        return 0, fmt.Errorf("createUser: %w", err)
    }

    id, err := result.LastInsertId() // AUTO_INCREMENT ID
    if err != nil {
        return 0, err
    }
    return id, nil
}

// UPDATE — RowsAffected 활용
func updateUserEmail(db *sql.DB, id int, newEmail string) (int64, error) {
    result, err := db.Exec(
        "UPDATE users SET email = ? WHERE id = ?",
        newEmail, id,
    )
    if err != nil {
        return 0, fmt.Errorf("updateUserEmail: %w", err)
    }

    affected, err := result.RowsAffected() // 변경된 행 수
    if err != nil {
        return 0, err
    }
    return affected, nil
}

// DELETE
func deleteUser(db *sql.DB, id int) error {
    result, err := db.Exec("DELETE FROM users WHERE id = ?", id)
    if err != nil {
        return fmt.Errorf("deleteUser: %w", err)
    }
    if n, _ := result.RowsAffected(); n == 0 {
        return fmt.Errorf("user id=%d not found", id)
    }
    return nil
}
Bulk INSERT (다중 행 삽입)
bulk.go go 
import (
    "fmt"
    "strings"
)

func bulkInsertUsers(db *sql.DB, users []User) error {
    if len(users) == 0 {
        return nil
    }

    // VALUES 플레이스홀더 동적 생성
    valueStrings := make([]string, 0, len(users))
    valueArgs   := make([]interface{}, 0, len(users)*3)

    for _, u := range users {
        valueStrings = append(valueStrings, "(?, ?, ?)")
        valueArgs   = append(valueArgs, u.Name, u.Email, u.Age)
    }

    query := fmt.Sprintf(
        "INSERT INTO users (name, email, age) VALUES %s",
        strings.Join(valueStrings, ","),
    )

    _, err := db.Exec(query, valueArgs...)
    return err
}
06

Prepared Statement

Prepared Statement는 동일한 쿼리를 반복 실행할 때 성능을 높입니다. MySQL 서버가 쿼리를 한 번 파싱·컴파일하고, 이후 실행 시 바인딩된 파라미터만 전달합니다. SQL Injection도 원천 차단됩니다.

stmt.go go 
type UserUpdate struct {
    ID   int
    Name string
    Age  int
}

func batchUpdate(db *sql.DB, updates []UserUpdate) error {
    // Prepare: 쿼리를 미리 컴파일
    stmt, err := db.Prepare("UPDATE users SET name = ?, age = ? WHERE id = ?")
    if err != nil {
        return fmt.Errorf("prepare: %w", err)
    }
    defer stmt.Close() // 반드시 닫기

    for _, upd := range updates {
        // 컴파일된 쿼리에 파라미터만 바인딩해 실행
        if _, err := stmt.Exec(upd.Name, upd.Age, upd.ID); err != nil {
            return fmt.Errorf("exec id=%d: %w", upd.ID, err)
        }
    }
    return nil
}

// Stmt를 이용한 SELECT
func preparedSelect(db *sql.DB) error {
    stmt, err := db.Prepare("SELECT id, name FROM users WHERE status = ? LIMIT ?")
    if err != nil { return err }
    defer stmt.Close()

    rows, err := stmt.Query("active", 100)
    if err != nil { return err }
    defer rows.Close()

    for rows.Next() {
        var id int
        var name string
        rows.Scan(&id, &name)
        fmt.Printf("%d: %s\n", id, name)
    }
    return rows.Err()
}
💡
Stmt와 트랜잭션
트랜잭션 내에서 Prepared Statement를 사용하려면 tx.Stmt(stmt)로 기존 stmt를 트랜잭션 범위로 변환하거나, tx.Prepare()로 새로 생성해야 합니다.

Named Parameter

트랜잭션 (Transaction)

트랜잭션은 여러 쿼리를 하나의 원자적 단위로 묶어 ACID 속성을 보장합니다. Go의 sql.Tx는 트랜잭션 전용 커넥션을 단독으로 사용합니다.

기본 트랜잭션 패턴
transaction.go go 
// 계좌 이체 예제 — 원자성 보장
func transfer(db *sql.DB, fromID, toID, amount int) error {
    // 트랜잭션 시작
    tx, err := db.Begin()
    if err != nil {
        return fmt.Errorf("begin tx: %w", err)
    }

    // defer로 롤백 보장 — Commit 이후엔 no-op이 됨
    defer func() {
        if p := recover(); p != nil {
            tx.Rollback()
            panic(p)
        }
    }()

    // 출금 계좌에서 차감
    res, err := tx.Exec(
        "UPDATE accounts SET balance = balance - ? WHERE id = ? AND balance >= ?",
        amount, fromID, amount,
    )
    if err != nil {
        tx.Rollback()
        return fmt.Errorf("debit: %w", err)
    }
    if n, _ := res.RowsAffected(); n == 0 {
        tx.Rollback()
        return fmt.Errorf("잔액 부족 또는 계좌 없음")
    }

    // 입금 계좌에 추가
    if _, err = tx.Exec(
        "UPDATE accounts SET balance = balance + ? WHERE id = ?",
        amount, toID,
    ); err != nil {
        tx.Rollback()
        return fmt.Errorf("credit: %w", err)
    }

    // 트랜잭션 로그 기록
    if _, err = tx.Exec(
        "INSERT INTO tx_log (from_id, to_id, amount) VALUES (?, ?, ?)",
        fromID, toID, amount,
    ); err != nil {
        tx.Rollback()
        return fmt.Errorf("log: %w", err)
    }

    // 모든 작업 성공 시 커밋
    return tx.Commit()
}
헬퍼 함수로 트랜잭션 추상화
tx_helper.go go 
// 재사용 가능한 트랜잭션 래퍼
// named return(err)을 사용해 Commit 실패 시에도 defer Rollback이 정상 동작
func WithTx(ctx context.Context, db *sql.DB, fn func(*sql.Tx) error) (err error) {
    tx, err := db.BeginTx(ctx, nil)
    if err != nil {
        return
    }

    defer func() {
        if p := recover(); p != nil {
            _ = tx.Rollback()
            panic(p)
        } else if err != nil {
            // fn() 또는 Commit() 실패 시 named return err != nil → 자동 롤백
            _ = tx.Rollback()
        }
    }()

    if err = fn(tx); err != nil {
        return
    }
    // Commit 결과도 named return err에 할당됨
    err = tx.Commit()
    return
}

// 사용 예시
type OrderItem struct {
    ProductID int
    Qty       int
}

type Order struct {
    UserID int
    Total  float64
    Items  []OrderItem
}

func createOrderWithItems(db *sql.DB, order Order) error {
    return WithTx(context.Background(), db, func(tx *sql.Tx) error {
        res, err := tx.Exec(
            "INSERT INTO orders (user_id, total) VALUES (?, ?)",
            order.UserID, order.Total,
        )
        if err != nil { return err }

        orderID, _ := res.LastInsertId()
        for _, item := range order.Items {
            if _, err := tx.Exec(
                "INSERT INTO order_items (order_id, product_id, qty) VALUES (?, ?, ?)",
                orderID, item.ProductID, item.Qty,
            ); err != nil { return err }
        }
        return nil
    })
}

func txWithIsolation(ctx context.Context, db *sql.DB) error {
    tx, err := db.BeginTx(ctx, &sql.TxOptions{
        Isolation: sql.LevelSerializable, // 가장 엄격
        ReadOnly:  false,
    })
    if err != nil { return err }
    defer tx.Rollback()
    // ... 쿼리 실행 ...
    return tx.Commit()
}

// sql.LevelDefault           — 드라이버 기본값

// sql.LevelReadCommitted     — Committed 데이터만 읽기
// sql.LevelRepeatableRead    — MySQL 기본값, Phantom Read 방지
// sql.LevelSerializable      — 완전한 직렬화

// 읽기 전용 트랜잭션 (성능 최적화)
func txReadOnly(ctx context.Context, db *sql.DB) error {
    tx, err := db.BeginTx(ctx, &sql.TxOptions{
        Isolation: sql.LevelRepeatableRead,
        ReadOnly:  true, // MySQL이 읽기 최적화 수행
    })
    if err != nil { return err }
    defer tx.Rollback()
    // ... 읽기 쿼리 실행 ...
    return tx.Commit()
}

MySQL의 NULL 값은 Go의 기본 타입(string, int 등)으로 직접 Scan할 수 없습니다. sql.Null* 타입이나 포인터를 사용해야 합니다.

type Profile struct {
    ID       int
    Name     string
    Bio      sql.NullString   // nullable VARCHAR

    Score    sql.NullFloat64  // nullable FLOAT
    Verified sql.NullBool    // nullable BOOLEAN

func getProfile(db *sql.DB, id int) (*Profile, error) {
    var p Profile
    err := db.QueryRow(
        `SELECT id, name, bio, age, score, verified, birthday
           FROM profiles WHERE id = ?`, id,
    ).Scan(&p.ID, &p.Name, &p.Bio, &p.Age,
           &p.Score, &p.Verified, &p.Birthday)
    if err != nil { return nil, err }
    return &p, nil
}

Row 스캔 패턴

Columns()로 동적 스캔
scan.go go 
// 컬럼명을 모를 때 동적으로 결과를 map으로 변환
func queryToMap(db *sql.DB, query string, args ...interface{}) ([]map[string]interface{}, error) {
    rows, err := db.Query(query, args...)
    if err != nil { return nil, err }
    defer rows.Close()

    cols, err := rows.Columns() // 컬럼명 가져오기
    if err != nil { return nil, err }

    var result []map[string]interface{}

    for rows.Next() {
        // 각 컬럼에 대한 interface{} 슬라이스 생성
        vals := make([]interface{}, len(cols))
        ptrs := make([]interface{}, len(cols))
        for i := range vals { ptrs[i] = &vals[i] }

        if err := rows.Scan(ptrs...); err != nil { return nil, err }

        row := make(map[string]interface{})
        for i, col := range cols {
            row[col] = vals[i]
        }
        result = append(result, row)
    }
    return result, rows.Err()
}

// ColumnTypes()로 타입 정보 조회
func printColumnInfo(rows *sql.Rows) {
    types, _ := rows.ColumnTypes()
    for _, ct := range types {
        nullable, _ := ct.Nullable()
        fmt.Printf("col=%s dbType=%s nullable=%v\n",
            ct.Name(), ct.DatabaseTypeName(), nullable)
    }
}

Context & 타임아웃

Go 1.8부터 database/sql의 모든 주요 메서드에 Context를 받는 *Context 버전이 추가되었습니다. 타임아웃, 취소, HTTP 요청 생명주기와 쿼리를 연동할 때 필수입니다.

context.go go 
// 쿼리 타임아웃 설정
func queryWithTimeout(db *sql.DB) (string, error) {
    ctx, cancel := context.WithTimeout(context.Background(), 3*time.Second)
    defer cancel() // 반드시 cancel 호출

    var name string
    err := db.QueryRowContext(ctx,
        "SELECT name FROM users WHERE id = ?", 1,
    ).Scan(&name)
    return name, err
}

// HTTP 핸들러에서 요청 컨텍스트 전파
func UserHandler(w http.ResponseWriter, r *http.Request) {
    // r.Context()는 클라이언트 연결 끊기면 자동 취소
    ctx := r.Context()

    rows, err := db.QueryContext(ctx,
        "SELECT id, name FROM users ORDER BY created_at DESC LIMIT 20",
    )
    if err != nil {
        if errors.Is(err, context.Canceled) {
            http.Error(w, "request canceled", http.StatusRequestTimeout)
            return
        }
        http.Error(w, err.Error(), http.StatusInternalServerError)
        return
    }
    defer rows.Close()
    // ... rows 처리
}

// 트랜잭션 + Context
func txWithDeadline(db *sql.DB) error {
    ctx, cancel := context.WithDeadline(
        context.Background(),
        time.Now().Add(10*time.Second),
    )
    defer cancel()

    tx, err := db.BeginTx(ctx, nil)
    if err != nil { return err }
    defer tx.Rollback()

    if _, err = tx.ExecContext(ctx, "UPDATE ..."); err != nil {
        return err
    }
    return tx.Commit()
}

에러 처리

표준 에러 타입
error.go go 
import (
    "errors"
    "database/sql"
    "github.com/go-sql-driver/mysql"
)

func handleDBError(err error) {
    if err == nil { return }

    // 1. 결과 없음
    if errors.Is(err, sql.ErrNoRows) {
        fmt.Println("데이터 없음")
        return
    }

    // 2. 커넥션 풀 닫힘
    if errors.Is(err, sql.ErrConnDone) {
        fmt.Println("DB 커넥션 종료됨")
        return
    }

    // 3. 트랜잭션 종료됨
    if errors.Is(err, sql.ErrTxDone) {
        fmt.Println("트랜잭션 이미 종료")
        return
    }

    // 4. MySQL 드라이버 에러 분석
    var mysqlErr *mysql.MySQLError
    if errors.As(err, &mysqlErr) {
        switch mysqlErr.Number {
        case 1062: // ER_DUP_ENTRY
            fmt.Println("중복 키 에러:", mysqlErr.Message)
        case 1216, 1217: // FK 제약 위반
            fmt.Println("외래키 제약 위반")
        case 1205: // Lock wait timeout
            fmt.Println("Lock 대기 타임아웃")
        case 1213: // Deadlock
            fmt.Println("데드락 발생 — 재시도 필요")
        default:
            fmt.Printf("MySQL 에러 #%d: %s\n",
                mysqlErr.Number, mysqlErr.Message)
        }
        return
    }

    // 5. Context 취소/타임아웃
    if errors.Is(err, context.DeadlineExceeded) {
        fmt.Println("쿼리 타임아웃")
        return
    }

    fmt.Println("알 수 없는 에러:", err)
}
데드락 재시도 패턴
retry.go go 
const errDeadlock = 1213

func isDeadlock(err error) bool {
    var me *mysql.MySQLError
    return errors.As(err, &me) && me.Number == errDeadlock
}

func retryOnDeadlock(maxRetries int, fn func() error) error {
    var err error
    for i := 0; i < maxRetries; i++ {
        err = fn()
        if err == nil || !isDeadlock(err) {
            return err
        }
        // 지수 백오프로 재시도
        time.Sleep(time.Duration(1<i) * 10 * time.Millisecond)
        fmt.Printf("데드락 감지, 재시도 %d/%d\n", i+1, maxRetries)
    }
    return fmt.Errorf("최대 재시도 초과: %w", err)
}

Best Practices

완전한 프로덕션 패턴 예제
repository.go go 
// Repository 패턴으로 DB 로직 캡슐화
type UserRepository struct {
    db      *sql.DB
    stmtGet *sql.Stmt
}

func NewUserRepository(db *sql.DB) (*UserRepository, error) {
    stmt, err := db.Prepare("SELECT id, name, email FROM users WHERE id = ?")
    if err != nil { return nil, err }
    return &UserRepository{db: db, stmtGet: stmt}, nil
}

func (r *UserRepository) Close() { r.stmtGet.Close() }

func (r *UserRepository) GetByID(ctx context.Context, id int) (*User, error) {
    var u User
    err := r.stmtGet.QueryRowContext(ctx, id).Scan(&u.ID, &u.Name, &u.Email)
    if errors.Is(err, sql.ErrNoRows) {
        return nil, fmt.Errorf("user %d: not found", id)
    }
    if err != nil {
        return nil, fmt.Errorf("GetByID %d: %w", id, err)
    }
    return &u, nil
}

func (r *UserRepository) Create(ctx context.Context, u *User) (int64, error) {
    res, err := r.db.ExecContext(ctx,
        "INSERT INTO users (name, email) VALUES (?, ?)",
        u.Name, u.Email,
    )
    if err != nil {
        var me *mysql.MySQLError
        if errors.As(err, &me) && me.Number == 1062 {
            return 0, fmt.Errorf("email already exists")
        }
        return 0, fmt.Errorf("Create: %w", err)
    }
    return res.LastInsertId()
}
핵심 체크리스트
  • sql.DB는 전역으로 재사용 — 요청마다 Open/Close하지 않습니다. 한 번 Open 후 프로그램 전체에서 공유합니다.
  • 항상 defer rows.Close() — Query() 호출 직후 반드시 작성. 커넥션 누수 방지의 핵심입니다.
  • rows.Err() 확인 필수 — rows.Next() 반복 후 반드시 rows.Err()를 체크해 중간 에러를 감지합니다.
  • 플레이스홀더(?) 사용 — 문자열 포매팅으로 쿼리를 구성하지 않습니다. SQL Injection의 원인이 됩니다.
  • Context 버전 메서드 사용 — 모든 쿼리에 Context를 전달해 타임아웃과 취소를 지원합니다.
  • 트랜잭션에서 defer Rollback — Commit 성공 이후엔 no-op이 되므로 항상 defer로 안전망을 유지합니다.
  • 커넥션 풀 튜닝 — MaxOpenConns를 MySQL max_connections의 80% 이하로, ConnMaxLifetime을 wait_timeout 이하로 설정합니다.
  • 에러 래핑 — fmt.Errorf("context: %w", err)로 에러 컨텍스트를 유지합니다.
